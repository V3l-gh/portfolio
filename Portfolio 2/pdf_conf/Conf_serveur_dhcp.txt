# Démarrer Docker
sudo systemctl enable docker

docker --version


Docker pour Dnsmasq :

# Créer dossier
sudo mkdir -p /opt/dnsmasq/conf.d
sudo nano /opt/dhcp/dnsmasq.conf

# Dans dnsmasq.conf
interface=eno4
bind-interfaces

port=53
domain-needed
bogus-priv

no-resolv
server=1.1.1.1
server=8.8.8.8

local=/lan/
domain=lan
expand-hosts

dhcp-authoritative

dhcp-range=172.17.1.50,172.17.1.100,255.255.255.0,12h

dhcp-option=option:router,172.17.1.254
dhcp-option=option:dns-server,172.17.3.1
dhcp-option=option:domain-name,lan

log-dhcp
log-queries
log-facility=-


# Lancer Dnsmasq avec Docker
sudo docker run -d \
  --name dnsmasq \
  --restart unless-stopped \
  --net=host \
  --cap-add=NET_ADMIN \
  -v /opt/dnsmasq/dnsmasq.conf:/etc/dnsmasq.conf:ro \
  -v /opt/dnsmasq/conf.d:/etc/dnsmasq.d:ro \
  jpillora/dnsmasq

# Conf IP
sudo nmcli con mod eno4 \
  ipv4.method manual \
  ipv4.addresses 172.17.3.1/29 \
  ipv4.gateway 172.17.3.6 \
  ipv4.dns 127.0.0.1

sudo nmcli con up eno4

ip route

# Pour changer priorité route ip
sudo nmcli con mod eno4 ipv4.route-metric 50
sudo nmcli con up eno4

# Règles Firewalld du serveur
firewall-cmd --permanent --add-service=dns
firewall-cmd --permanent --add-service=dhcp
firewall-cmd --reload


Docker LDAP :

# Créer dossier
sudo mkdir -p /srv/docker/ldap/{data,config}

# Lancer image
docker run -d \
  --name ldap \
  -p 389:389 \
  -p 636:636 \
  -v /srv/docker/ldap/data:/var/lib/ldap \
  -v /srv/docker/ldap/config:/etc/ldap/slapd.d \
  -e LDAP_ORGANISATION="Ubisoft Lab" \
  -e LDAP_DOMAIN="ubisoft.lab" \
  -e LDAP_ADMIN_PASSWORD="admin01" \
  --restart always \
  osixia/openldap

# Règles Firewall
sudo firewall-cmd --permanent --add-port=389/tcp
sudo firewall-cmd --permanent --add-port=636/tcp
sudo firewall-cmd --reload
